70 inline MIME_TYPE mime_type ( const std::string & extension ) {
71  int position_ = 0;
72  std::string clean_ext_ = extension;

74  if ( clean_ext_ == "jpeg" ) { clean_ext_ = "jpg"; }

76  for ( mapping * m = mappings; m->extension; ++m ) {
77  if ( m->extension == clean_ext_ ) {
78  return MIME_TYPE ( position_ );
