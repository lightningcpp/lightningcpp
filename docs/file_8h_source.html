34 static inline http_status parse_filename ( const std::string & path, std::string & filename, std::string & extension ) {
35  std::size_t last_slash_pos = path.find_last_of ( "/" );
36  std::size_t last_dot_pos = path.find_last_of ( "." );
37 
38  if ( last_dot_pos != std::string::npos && last_dot_pos > last_slash_pos ) {
39  extension = path.substr ( last_dot_pos + 1 );
40 
41  } else { extension = "txt"; }
42 
43  if ( last_slash_pos != std::string::npos && last_slash_pos < path.size() ) {
44  filename = path.substr ( last_slash_pos + 1 );
45 
46  } else { return http_status::BAD_REQUEST; }
47  return http_status::OK;
48 }

73  http_status execute ( Request& request, Response& response ) {
74 
75  /* test if the path is valid */
76  if ( !valid_path ( request.uri() ) ) {
77  return http::http_status::BAD_REQUEST;
78  }
79 
80  std::stringstream ss_;
81  if( prefix_.empty() ) { //handle prefix
82  ss_ << docroot_ << request.uri();
83  } else {
84  ss_ << docroot_ << request.uri().substr( prefix_.size() );
85  }
86 
87  struct stat filestatus;
88  std::string extension_, filename_;
89  stat ( ss_.str().c_str(), &filestatus );
90 
91  //when the request path is a directory add index.html
92  if ( S_ISDIR ( filestatus.st_mode ) ) {
93  filename_ = "index.html";
94  extension_ = "html";
95  ss_ << "/index.html";
96  stat ( ss_.str().c_str(), &filestatus );
97 
98  } else {
99  //get the filename and extension
100  auto _status = _file_utils::parse_filename ( ss_.str(), filename_, extension_ );
101  if( _status != http_status::OK ) { //when the filename can not be parsed.
102  return _status;
103  }
104  }
105 
106  if ( S_ISREG ( filestatus.st_mode ) ) {
107  /* test if the file has changed */
108  std::string s_time_ = http::utils::time_to_string ( &filestatus.st_mtime );
109 
110  if ( request.contains_parameter ( http::header::IF_MODIFIED_SINCE ) &&
111  request.parameter ( http::header::IF_MODIFIED_SINCE ) == s_time_ ) {
112  return http_status::NOT_MODIFIED;
113  }
114 
115  //add headers
116  response.parameter ( header::CONTENT_LENGTH, std::to_string ( filestatus.st_size ) );
117  response.parameter ( http::header::CONTENT_DISPOSITION, "inline; filename= \"" + filename_ + "\"" );
118  response.parameter ( http::header::CONTENT_TYPE, mime::mime_type ( mime::mime_type ( extension_ ) ) );
119  response.last_modified ( filestatus.st_mtime );
120  response.expires ( 3600 * 24 );
121 
122  //process request
123  if ( request.method() == http::method::GET ) {
124  auto _status = get ( request, response, ss_.str(), static_cast< size_t >( filestatus.st_size ) );
125  if( _status != http_status::OK )
126  { return _status; }
127  } else if ( request.method() != http::method::HEAD )
128  { return http::http_status::NOT_IMPLEMENTED; }
129  } else { return http::http_status::NOT_FOUND; }
130  return http_status::OK;
131  }

135  http_status get ( Request & request, Response & response, const std::string & filename, const size_t file_size ) {
136 
137  // Open the file to send back.
138  std::unique_ptr < std::ifstream > is =
139  std::make_unique< std::ifstream > ( filename.c_str(), std::ios::in | std::ios::binary );
140 
141  if ( !is->is_open() ) {
142  return http_status::NOT_FOUND;
143  }
144 
145  // Fill out the reply to be sent to the client.
146  if ( request.contains_parameter ( http::header::RANGE ) ) {
147  response.status ( http_status::PARTIAL_CONTENT );
148  std::tuple<int, int> range = http::utils::parseRange ( request.parameter ( http::header::RANGE ) );
149  //std::cout << "\trange: " << std::get<0> ( range ) << "-" << std::get<1> ( range ) << std::endl;
150  response.parameter ( "Content-Range", "bytes " + std::to_string ( std::get<0> ( range ) ) + "-" +
151  ( std::get<1> ( range ) == -1 ? std::to_string ( file_size - 1 ) :
152  std::to_string ( std::get<1> ( range ) - 1 ) ) +
153  "/" + std::to_string ( file_size ) );
154  response.parameter ( header::CONTENT_LENGTH, ( std::get<1> ( range ) == -1 ?
155  std::to_string ( file_size - static_cast< size_t >( std::get<0> ( range ) ) ) :
156  std::to_string ( std::get<1> ( range ) - std::get<0> ( range ) ) ) );
157  is->seekg ( std::get<0> ( range ), std::ios_base::beg ); //TODO check if range is available
158 
159  } else {
160  response.parameter ( header::CONTENT_LENGTH, std::to_string ( file_size ) );
161  response.status ( http_status::OK );
162  }
163 
164  response.istream ( std::move ( is ) );
165  return http_status::OK;
166  }
