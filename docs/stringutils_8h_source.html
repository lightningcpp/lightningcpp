229 struct UrlParser {
230  UrlParser( const std::string& url ) {
231  size_t _position = 0;
232  if( url.find( "http://" ) != std::string::npos ) {
233  secure_ = false;
234  proto_ = "http";
235  _position = std::string( "http://" ).size();
236  } else if( url.find( "https://" ) != std::string::npos ) {
237  secure_ = true;
238  proto_ = "https";
239  _position = std::string( "https://" ).size();
240  } else { /*throw exception*/ }
241 
242  if( url.find( ":", _position ) != std::string::npos &&
243  url.find( ":", _position ) < url.find( "/", _position ) ) {
244  size_t _dash_pos = url.find( ":", _position );
245  size_t _end_port_pos = url.find( "/", _dash_pos );
246  if( _end_port_pos == std::string::npos ) _end_port_pos = url.size();
247  host_ = url.substr( _position, _dash_pos - _position );
248  proto_ = url.substr( _dash_pos + 1, _end_port_pos - _dash_pos - 1 );
249  _position = _end_port_pos;
250  } else {
251  size_t _slash_pos = url.find( "/", _position );
252  if( _slash_pos == std::string::npos ) _slash_pos = url.size();
253  host_ = url.substr( _position, _slash_pos - _position );
254  _position = _slash_pos;
255  }
256 
257  if( _position == url.size() ) {
258  path_ = "/";
259  } else {
260  path_ = url.substr( _position, url.size() - _position );
261  }
262  }
263 
264  std::string host ()
265  { return host_; }
266  std::string proto ()
267  { return proto_; }
268  std::string path ()
269  { return path_; }
270  bool secure ()
271  { return secure_; }

273 private:
