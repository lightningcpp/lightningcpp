99  // create status line: Status-Line = HTTP-Version SP Status-Code SP Reason-Phrase CRLF
100  auto _position = snprintf ( buffer, static_cast< size_t >( size ), "%s/%d.%d %d %s\r\n",

102 
103  if ( _position < 0 && _position > size ) { throw http_status::INTERNAL_SERVER_ERROR; } /* TODO */
104

106  _position += snprintf ( buffer+_position,
107  static_cast< size_t >( size ) - static_cast< size_t >( _position ),
